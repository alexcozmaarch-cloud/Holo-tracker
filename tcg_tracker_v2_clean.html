⚙ Setup Guide
3 minutes · free · no coding needed
1
Create a Google Sheet
Go to sheets.google.com → create a blank spreadsheet → name it HOLO TCG Tracker.
2
Open Apps Script and paste the code
In your sheet click Extensions → Apps Script. Delete everything and paste this code:
const SHEET_NAME = 'Cards'; function doPost(e) { try { const payload = JSON.parse(e.postData.contents); const ss = SpreadsheetApp.getActiveSpreadsheet(); let sheet = ss.getSheetByName(SHEET_NAME); if (!sheet) { sheet = ss.insertSheet(SHEET_NAME); sheet.appendRow(['id','name','set','game','notes','firstSeen','mentions','priceHistory']); } if (payload.action === 'getAll') return respond(getAllCards(sheet)); if (payload.action === 'upsert') { upsertCard(sheet, payload.card); return respond(getAllCards(sheet)); } if (payload.action === 'delete') { deleteCard(sheet, payload.id); return respond(getAllCards(sheet)); } return respond({ error: 'Unknown action' }); } catch(err) { return respond({ error: err.message }); } } function getAllCards(sheet) { const rows = sheet.getDataRange().getValues(); if (rows.length <= 1) return { cards: [] }; return { cards: rows.slice(1).map(r => ({ id: r[0], name: r[1], set: r[2], game: r[3], notes: r[4], firstSeen: r[5], mentions: r[6], priceHistory: JSON.parse(r[7] || '[]') }))}; } function upsertCard(sheet, card) { const rows = sheet.getDataRange().getValues(); for (let i = 1; i < rows.length; i++) { if (rows[i][0] == card.id) { sheet.getRange(i+1,1,1,8).setValues([[ card.id, card.name, card.set, card.game||'', card.notes||'', card.firstSeen, card.mentions, JSON.stringify(card.priceHistory) ]]); return; } } sheet.appendRow([ card.id, card.name, card.set, card.game||'', card.notes||'', card.firstSeen, card.mentions, JSON.stringify(card.priceHistory) ]); } function deleteCard(sheet, id) { const rows = sheet.getDataRange().getValues(); for (let i = rows.length - 1; i >= 1; i--) { if (rows[i][0] == id) { sheet.deleteRow(i + 1); return; } } } function respond(data) { return ContentService .createTextOutput(JSON.stringify(data)) .setMimeType(ContentService.MimeType.JSON); }
3
Deploy as a Web App
Click Deploy → New deployment → click the ⚙️ gear → select Web app
— Execute as: Me
— Who has access: Anyone
Click Deploy → when Google asks to authorize click Advanced → Go to project (unsafe) → Allow
Copy the URL ending in /exec
4
Paste URL and connect
Paste the URL into the Connect box at the top → click Connect. Done ✅
Next time you open this file it connects automatically.
